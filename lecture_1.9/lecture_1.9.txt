1. выбрать рейсы (несколько полей, включая дата старта и название) со стартом в 2020 году

SELECT Name,Start_Date__c FROM Trip__c WHERE Start_Date__c >= 2020-01-01 AND Start_Date__c <= 2020-12-31

2. выбрать туристов (имя, возраст) чье имя содержит Джон

SELECT Age__c,Name FROM Tourist__c WHERE Name LIKE '%Джон%'

3. выбрать туристов (имя, возраст) с возрастом 20, 22, 24 года

SELECT Age__c,Name FROM Tourist__c WHERE Age__c = 20 OR Age__c = 22 OR Age__c = 24

4. выбрать рейсы (дата старта, название) со списком туристов (имя, возраст)
SELECT Trip__r.Start_Date__c, Trip__r.Name, Tourist__r.Name, Tourist__r.Age__c FROM Flight__c

5. выбрать туристов (имя, возраст) со списком рейсов (дата старта, название)

SELECT Tourist__r.Name, Tourist__r.Age__c, Trip__r.Start_Date__c, Trip__r.Name  FROM Flight__c

6. выбрать туристов (имя, возраст) которые не зарегистрировались ни на один рейс

SELECT Age__c,Name FROM Tourist__c WHERE Id NOT IN (SELECT Tourist__c FROM Flight__c)
